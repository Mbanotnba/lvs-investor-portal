Quick Start Example

# LVS SDK - Deploy Any Vision Model
import lvs
from lvs.vision import Camera, ImageProcessor
from lvs.inference import Runtime

# Initialize runtime on LVS-250
runtime = lvs.Runtime(device="lvs250")

# Load YOUR compiled model (any ONNX, PyTorch, or TensorFlow model)
model = runtime.load_model("your_model.lvs")

# Real-time inference loop
for frame in Camera.stream(channel=0):
    results = model.infer(frame)
    print(f"Inference: {model.last_inference_ms:.1f}ms")
